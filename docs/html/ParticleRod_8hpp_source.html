8 // generating max 2 contacts if they stray too far apart or are too close

10 public:
11 ParticleRod(float rodLenght, Particle* p0, Particle* p1): ParticleLink(p0, p1), lenght(rodLenght) {}

13 // Fill in given contact pointer with contact needed to prevent the rod from overextending or compressing
14 virtual unsigned addContact(ParticleContact* contact, unsigned limit) const override {
15 float currentLength = getCurrentLenght();
16
17 // Check if rod is overextended
18 if (currentLength == lenght) return 0;
19
20 // Otherwise, get the contact
21 contact->particle[0] = particle[0];
22 contact->particle[1] = particle[1];
23
24 // Calculate the normal
25 glm::vec3 normal = glm::normalize(particle[1]->getPosition() - particle[0]->getPosition());
26
27 // Normal depends on whether rod is overextending or compressing
28 if (currentLength > lenght) {
29 contact->contactNormal = normal;
30 contact->penetration = currentLength - lenght;
31 } else {
32 contact->contactNormal = normal * -1.0f;
33 contact->penetration = lenght - currentLength;
34 }
35
36 // No bounciness for rods so restituion is always zero!
37 contact->restitution = 0.0f;
38
39 return 1;
40
41 }
