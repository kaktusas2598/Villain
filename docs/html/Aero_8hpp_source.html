20 // Exactly the same as updateForce, except that this takes explicit tensor matrix
21 void updateForceFromTensor(RigidBody* body, float deltaTime, const glm::mat3& aeroTensor) {
22 // Calculate total velocity (wind speed and body's velocity)
23 glm::vec3 velocity = body->getLinearVelocity();
24 velocity += *windSpeed;
25
26 // Calculate the velocity in body coordinates
27 glm::vec3 bodyVel = glm::inverse(body->getTransform()) * glm::vec4(velocity, 1.0);
28
29 // Calculate the force in body coordinates
30 glm::vec3 bodyForce = aeroTensor * bodyVel;
31 glm::vec3 force = body->getTransform() * glm::vec4(bodyForce, 1.0);
32
33 // Finally apply the force
34 body->addForceAtBodyPoint(force, position);
35 }
36
37 glm::mat3 tensor; //< Aerodynamic tensor for surface in body/local space
38 glm::vec3 position; //< Relative position of the aerodynamic surface in body coords
39 glm::vec3* windSpeed; //< Pointer to the wind speed of the environment, that way it can be controlled globally
40 };
