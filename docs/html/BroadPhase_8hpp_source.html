108  template <class BoundingVolumeClass>
109  unsigned BVHNode<BoundingVolumeClass>::getPotentialContactsWith(
110  const BVHNode<BoundingVolumeClass>* other, PotentialContact* contacts, unsigned limit) const {
111  if (!overlaps(other) || limit == 0) return 0;
112 
113  // If both are leaf nodes then there is potential contact
114  if (isLeaf() && other->isLeaf()) {
115  contacts->body[0] = body;
116  contacts->body[1] = other->body;
117  return 1;
118  }
119 
120  // Determine which node to descend into - if either node is leaf, then descend into the other.
121  // If both nodes are branches, then descend into bigger one
122  if (other->isLeaf() || (!isLeaf() && volume->getSize() >= other->volume->getSize())) {
123  // Recurse into self
124  unsigned count = children[0]->getPotentialContactsWith(other, contacts, limit);
125 
126  // Check if there are enough slots left to check other node
127  if (limit > count) {
128  return count + children[1]->getPotentialContactsWith(other, contacts + count, limit - count);
129  } else {
130  return count;
131  }
132  } else {
133  // Recurse into other node
134  unsigned count = getPotentialContactsWith(other->children[0], contacts, limit);
135 
136  // Check if there are enough slots left to check other node
137  if (limit > count) {
138  return count + other->children[1]->getPotentialContactsWith(other, contacts + count, limit - count);
139  } else {
140  return count;
141  }
142  }

145  template <class BoundingVolumeClass>
146  void BVHNode<BoundingVolumeClass>::insert(RigidBody* newBody, const BoundingVolumeClass& newVolume) {
147  // if this node is a leaf, then the only option is to create two new children and place the body in one
148  if (isLeaf()) {
149  // Child one is copy of this node
150  children[0] = new BVHNode<BoundingVolumeClass>(this, volume, body);
151 
152  // Child two will hold the new node
153  children[0] = new BVHNode<BoundingVolumeClass>(this, newVolume, newBody);
154 
155  // Make sure this node is not a leaf anymore
156  this->body = nullptr;
157 
158  recalculateBoundingVolume();
159  } else {
160  // Otherwise work out which child gets to keep the inserted body, give it to whoever would grow
161  // the least to incorporate this new body
162 
163  if (children[0]->volume.getGrowth(newVolume) < children[1]->volume.getGrowth(newVolume)) {
164  children[0]->insert(newBody, newVolume);
165  } else {
166  children[1]->insert(newBody, newVolume);
167  }
168  }
